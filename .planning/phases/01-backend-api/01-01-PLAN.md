--- cleanHTML() FUNCTION ---

The function receives a cheerio $ object (already loaded), mutates it in-place, and returns { $, removed, vslDetected }.

```js
function cleanHTML($) {
  let removed = 0;
  let vslDetected = false;

  // Helper: does a string match any keyword? (case-insensitive)
  function matchesAny(str, keywords) {
    if (!str) return false;
    const lower = str.toLowerCase();
    return keywords.some(k => lower.includes(k.toLowerCase()));
  }

  // Helper: does a string match any preserve keyword?
  function shouldPreserve(str) {
    return matchesAny(str, PRESERVE_KEYWORDS);
  }

  // CLEAN-01 + CLEAN-02: Remove <script> tags
  // Check: src attribute OR inline text content contains removal keyword
  // AND does NOT match any preserve keyword
  // Also remove if element has data-vturb or data-player-id attribute
  $('script').each(function () {
    const el   = $(this);
    const src  = el.attr('src') || '';
    const text = el.html() || '';
    const hasVturbAttr = el.attr('data-vturb') !== undefined || el.attr('data-player-id') !== undefined;

    if (shouldPreserve(src) || shouldPreserve(text)) return; // CLEAN-07

    if (hasVturbAttr || matchesAny(src, SCRIPT_REMOVE_KEYWORDS) || matchesAny(text, SCRIPT_REMOVE_KEYWORDS)) {
      if (src.toLowerCase().includes('vturb') || hasVturbAttr) vslDetected = true;
      el.remove();
      removed++;
    }
  });

  // CLEAN-03: Remove <noscript> blocks
  // Remove if their text content contains any NOSCRIPT_REMOVE_KEYWORDS
  // OR if they contain <img> tags (typically tracking pixels)
  $('noscript').each(function () {
    const el   = $(this);
    const html = el.html() || '';
    if (matchesAny(html, NOSCRIPT_REMOVE_KEYWORDS) || html.includes('<img')) {
      el.remove();
      removed++;
    }
  });

  // CLEAN-04: Remove <link rel="preload"> or <link rel="prefetch"> pointing to video or vturb
  $('link[rel="preload"], link[rel="prefetch"]').each(function () {
    const el   = $(this);
    const href = el.attr('href') || '';
    if (
      PRELOAD_REMOVE_EXTENSIONS.some(ext => href.toLowerCase().endsWith(ext)) ||
      href.toLowerCase().includes(PRELOAD_REMOVE_DOMAIN)
    ) {
      el.remove();
      removed++;
    }
  });

  // CLEAN-05 + CLEAN-06: Remove player iframes and their wrapper divs
  // Strategy:
  //   1. For each matching iframe, attempt to remove the closest wrapping div
  //      if that div has a class/id containing a PLAYER_DIV_KEYWORD or
  //      "video-container" AND the div has no other meaningful children besides
  //      the iframe and the placeholder comment.
  //   2. Inject VSL_PLACEHOLDER at the position where the removed element was.
  //   3. If no wrapper found, inject placeholder directly after the iframe, then remove the iframe.
  $('iframe').each(function () {
    const el  = $(this);
    const src = el.attr('src') || '';
    if (!TRACKING_IFRAME_DOMAINS.some(domain => src.includes(domain))) return;

    vslDetected = true;

    // Look up the DOM for a wrapper div
    let target = el;
    let parent = el.parent();
    while (parent.length && parent[0].tagName && parent[0].tagName.toLowerCase() !== 'body') {
      const cls = (parent.attr('class') || '').toLowerCase();
      const id  = (parent.attr('id')    || '').toLowerCase();
      const combined = cls + ' ' + id;
      if (
        PLAYER_DIV_KEYWORDS.some(k => combined.includes(k)) ||
        combined.includes('video-container')
      ) {
        target = parent;
        break;
      }
      parent = parent.parent();
    }

    target.after(VSL_PLACEHOLDER);
    target.remove();
    removed++;
  });

  return { $, removed, vslDetected };
}
```

--- detectCheckout() FUNCTION ---

```js
function detectCheckout($) {
  const links = [];
  const seen  = new Set();

  $('a[href], button[onclick]').each(function (i) {
    const el      = $(this);
    const href    = el.attr('href') || '';
    const onclick = el.attr('onclick') || '';
    const combined = href + onclick;

    // Must match at least one checkout pattern
    const match = CHECKOUT_PATTERNS.find(cp => combined.toLowerCase().includes(cp.pattern.toLowerCase()));
    if (!match) return;

    // Deduplicate by href
    const key = href || onclick;
    if (seen.has(key)) return;
    seen.add(key);

    // Build a CSS selector unique enough to re-find this element at export time.
    // Strategy: use tag + nth-of-type among all elements with the same tag.
    // This is reliable because the HTML structure won't change between fetch and export.
    const tag       = this.tagName.toLowerCase();
    const allOfTag  = $(tag);
    const idx       = allOfTag.index(this); // 0-based
    const selector  = `${tag}:nth-of-type(${idx + 1})`;

    // Determine bundle from surrounding text context (parent + siblings text)
    const contextText = (el.closest('div, section, td, li').text() || el.text() || '').trim();
    let bundle = null;
    for (const sig of BUNDLE_SIGNALS) {
      if (sig.regex.test(contextText)) { bundle = sig.bundle; break; }
    }

    links.push({
      originalHref: href || onclick,
      selector,
      labelText: el.text().trim().slice(0, 120),
      platform: match.platform,
      bundle,
    });
  });

  return links;
}
```

--- POST /api/fetch ROUTE ---

```js
app.post('/api/fetch', async (req, res) => {
  const { url } = req.body;
  if (!url || typeof url !== 'string') {
    return res.status(400).json({ error: 'url is required' });
  }

  let rawHtml;
  try {
    const response = await axios.get(url, {
      headers: FETCH_HEADERS,
      maxRedirects: 10,
      responseType: 'text',
      timeout: 30000,
    });
    rawHtml = response.data;
  } catch (err) {
    return res.status(502).json({ error: `Fetch failed: ${err.message}` });
  }

  const $ = cheerio.load(rawHtml, { decodeEntities: false });

  const { removed, vslDetected } = cleanHTML($);
  const checkoutLinks = detectCheckout($);

  const html = $.html();

  return res.json({
    html,
    summary: {
      scriptsRemoved: removed,
      vslDetected,
      checkoutLinks,
    },
  });
});
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user → POST /api/fetch | User-supplied URL is fetched server-side; SSRF risk |
| remote VSL page → server | Attacker-controlled HTML is parsed but never eval'd |
| POST /api/export body | Large or malformed HTML payloads |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | /api/fetch url param | accept | App is single-user local; no auth needed |
| T-01-02 | Tampering | axios response body | accept | cheerio parses as HTML string; no eval or innerHTML on server |
| T-01-03 | Information Disclosure | SSRF via url param | mitigate | Validate url starts with http:// or https:// before axios.get; reject file://, ftp://, internal IPs |
| T-01-04 | Denial of Service | Large response body | mitigate | Set axios maxContentLength: 10 * 1024 * 1024 (10 MB limit) |
| T-01-05 | Elevation of Privilege | N/A | accept | No auth layer; single-user local app; no privilege escalation surface |
