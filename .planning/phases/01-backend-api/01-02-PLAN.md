Also add these two mitigations to /api/fetch (they were called out in plan 01-01 threat model). Insert them at the TOP of the `/api/fetch` handler, after the `url` extraction:

```js
  // T-01-03 mitigation: reject non-http(s) schemes and private IP ranges
  try {
    const parsed = new URL(url);
    if (!['http:', 'https:'].includes(parsed.protocol)) {
      return res.status(400).json({ error: 'Only http and https URLs are allowed' });
    }
    // Block private / loopback addresses
    const hostname = parsed.hostname.toLowerCase();
    if (
      hostname === 'localhost' ||
      hostname === '127.0.0.1' ||
      hostname.startsWith('192.168.') ||
      hostname.startsWith('10.') ||
      hostname.startsWith('172.') ||
      hostname.endsWith('.local')
    ) {
      return res.status(400).json({ error: 'Private/loopback URLs are not allowed' });
    }
  } catch {
    return res.status(400).json({ error: 'Invalid URL' });
  }
```

Also add `maxContentLength: 10 * 1024 * 1024` (T-01-04) to the axios.get options in /api/fetch.

--- POST /api/export IMPLEMENTATION ---

Insert this route in server.js before app.listen:

```js
app.post('/api/export', (req, res) => {
  const { html, headerPixel, headerPreload, vslembed, checkoutLinks } = req.body;

  if (!html || typeof html !== 'string') {
    return res.status(400).json({ error: 'html is required' });
  }

  // 1. Re-parse with cheerio (decodeEntities: false to preserve original encoding)
  const $ = cheerio.load(html, { decodeEntities: false });

  // 2. EXPORT-02: Inject headerPixel + headerPreload before </head>
  //    Use cheerio to append to <head> so insertion is always valid.
  if (headerPixel && headerPixel.trim()) {
    $('head').append(headerPixel);
  }
  if (headerPreload && headerPreload.trim()) {
    $('head').append(headerPreload);
  }

  // 3. EXPORT-03: Replace <!-- [VSL_PLACEHOLDER] --> block with vslembed.
  //    Cheerio serialises comments; we operate on the raw HTML string after
  //    $.html() because cheerio cannot reliably select HTML comments.
  let outputHtml = $.html();

  if (vslembed && vslembed.trim()) {
    // Replace the entire placeholder div (from <!-- [VSL_PLACEHOLDER] --> through </div>)
    // Use a regex that matches the comment + the vsl-cloner-placeholder div block.
    outputHtml = outputHtml.replace(
      /<!--\s*\[VSL_PLACEHOLDER\]\s*-->[\s\S]*?<div id="vsl-cloner-placeholder"[\s\S]*?<\/div>/,
      vslembed
    );
  }

  // 4. EXPORT-04: Replace checkout link hrefs.
  //    Re-parse the post-embed HTML to apply selector-based replacements.
  if (Array.isArray(checkoutLinks) && checkoutLinks.length > 0) {
    const $2 = cheerio.load(outputHtml, { decodeEntities: false });
    checkoutLinks.forEach(({ selector, affiliateHref }) => {
      if (!selector || !affiliateHref) return;
      try {
        $2(selector).each(function () {
          const el = $2(this);
          // Replace href on <a>, or update onclick/href pattern on <button>
          if (this.tagName.toLowerCase() === 'a') {
            el.attr('href', affiliateHref);
          } else {
            // For buttons with onclick="location.href=..." pattern
            const current = el.attr('onclick') || '';
            if (current) {
              el.attr('onclick', current.replace(/https?:\/\/[^\s'"]+/, affiliateHref));
            }
            // Also set a data attribute for debugging
            el.attr('data-affiliate-href', affiliateHref);
          }
        });
      } catch (e) {
        // Invalid selector — skip silently, do not crash the export
        console.warn(`[export] invalid selector skipped: ${selector}`);
      }
    });
    outputHtml = $2.html();
  }

  // 5. EXPORT-05: Return as downloadable file
  res.setHeader('Content-Type', 'text/html; charset=utf-8');
  res.setHeader('Content-Disposition', 'attachment; filename="pagina-afiliado.html"');
  res.send(outputHtml);
});
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user → POST /api/export | User-supplied HTML (from a prior fetch) is re-parsed and returned |
| cheerio selector → DOM | Affiliate-supplied CSS selectors are evaluated inside cheerio |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | /api/export html body | accept | HTML is re-parsed by cheerio (not eval'd); no server-side script execution risk |
| T-02-02 | Denial of Service | /api/export large html | mitigate | express.json({ limit: '10mb' }) already set in plan 01-01; no additional action needed |
| T-02-03 | Injection | cheerio selector from checkoutLinks | mitigate | Wrap each $2(selector) call in try/catch; invalid selectors are skipped with console.warn |
| T-02-04 | Information Disclosure | static file serving | accept | public/ serves only intentionally placed files; no directory traversal via express.static defaults |
</threat_model>
