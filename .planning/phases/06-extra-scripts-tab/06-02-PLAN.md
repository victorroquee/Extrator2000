**Step 3 — Add helper functions for the extra scripts list.**

Add these functions in the helpers block (near `showSection`, `showError`, `clearError`):

```js
    // ── Extra Scripts helpers (Phase 6) ─────────────────────────────

    function renderScriptLabels() {
      // Re-number all visible script item labels and update ↑/↓ disabled states
      var items = scriptsListContainer.querySelectorAll('.script-item');
      items.forEach(function(item, idx) {
        var label = item.querySelector('.script-item-label');
        if (label) label.textContent = 'Script ' + (idx + 1);

        var btnUp   = item.querySelector('[data-action="up"]');
        var btnDown = item.querySelector('[data-action="down"]');
        if (btnUp)   btnUp.disabled   = (idx === 0);
        if (btnDown) btnDown.disabled = (idx === items.length - 1);

        // Update aria-labels for accessibility
        if (btnUp)   btnUp.setAttribute('aria-label', 'Mover script ' + (idx + 1) + ' para cima');
        if (btnDown) btnDown.setAttribute('aria-label', 'Mover script ' + (idx + 1) + ' para baixo');
        var btnRemove = item.querySelector('[data-action="remove"]');
        if (btnRemove) btnRemove.setAttribute('aria-label', 'Remover script ' + (idx + 1));
      });
    }

    function addScript() {
      // D-06: create a new script item and append to the list
      var id = (typeof crypto !== 'undefined' && crypto.randomUUID)
        ? crypto.randomUUID()
        : 'script-' + Date.now() + '-' + Math.random().toString(36).slice(2);

      state.extraScripts.push({ id: id, content: '' });

      var item = document.createElement('div');
      item.className = 'script-item';
      item.dataset.scriptId = id;

      item.innerHTML =
        '<div class="script-item-header">' +
          '<span class="script-item-label">Script ' + state.extraScripts.length + '</span>' +
          '<div class="script-item-controls">' +
            '<button type="button" class="btn-script-control" data-action="up" aria-label="Mover script para cima">\u2191</button>' +
            '<button type="button" class="btn-script-control" data-action="down" aria-label="Mover script para baixo">\u2193</button>' +
            '<button type="button" class="btn-script-control" data-action="remove" aria-label="Remover script">\u00d7</button>' +
          '</div>' +
        '</div>' +
        '<textarea class="mono" data-script-id="' + id + '" rows="4" ' +
          'placeholder="<!-- Cole aqui seu script extra -->\n<script>\n  // seu c\u00f3digo aqui\n</script>" ' +
          'style="min-height: 80px;"></textarea>';

      scriptsListContainer.appendChild(item);
      renderScriptLabels();

      // D-06: focus the new textarea
      var ta = item.querySelector('textarea');
      if (ta) ta.focus();
    }

    function removeScript(scriptId) {
      // D-07: remove item from DOM and state
      state.extraScripts = state.extraScripts.filter(function(s) { return s.id !== scriptId; });
      var item = scriptsListContainer.querySelector('[data-script-id="' + scriptId + '"]');
      if (!item) {
        // item is the .script-item wrapper, not the textarea
        item = scriptsListContainer.querySelector('.script-item[data-script-id="' + scriptId + '"]');
      }
      // Find the .script-item parent
      var scriptItem = scriptsListContainer.querySelector('.script-item[data-script-id="' + scriptId + '"]');
      if (scriptItem) scriptItem.remove();
      renderScriptLabels();
    }

    function moveScript(scriptId, direction) {
      // D-10/D-12: swap item with neighbor in both state array and DOM
      var idx = state.extraScripts.findIndex(function(s) { return s.id === scriptId; });
      if (idx === -1) return;

      var swapIdx = direction === 'up' ? idx - 1 : idx + 1;
      if (swapIdx < 0 || swapIdx >= state.extraScripts.length) return;

      // Swap in state
      var temp = state.extraScripts[idx];
      state.extraScripts[idx] = state.extraScripts[swapIdx];
      state.extraScripts[swapIdx] = temp;

      // Swap in DOM — get both .script-item nodes by data-script-id
      var allItems = Array.from(scriptsListContainer.querySelectorAll('.script-item'));
      var nodeA = allItems[idx];
      var nodeB = allItems[swapIdx];
      if (!nodeA || !nodeB) return;

      // Insert the earlier node after the later node (or vice versa)
      if (direction === 'up') {
        scriptsListContainer.insertBefore(nodeA, nodeB);
      } else {
        scriptsListContainer.insertBefore(nodeB, nodeA);
      }

      renderScriptLabels();
    }

    function buildExtraScriptsPayload() {
      // Read current textarea values into state and return string array in order
      return state.extraScripts.map(function(s) {
        var ta = scriptsListContainer.querySelector('textarea[data-script-id="' + s.id + '"]');
        return ta ? ta.value : s.content;
      });
    }
```
