### F. JavaScript — Upload handler

Adicionar DEPOIS do bloco de event listener do btnFetch (depois da linha ~1396, o `});` que fecha o btnFetch listener) e ANTES do urlInput keydown listener:

```javascript
// -- Folder upload --
var uploadInFlight = false;
btnUpload.addEventListener('click', async function() {
  var files = folderInput.files;
  if (!files || !files.length) {
    showError('Selecione uma pasta antes de processar.');
    return;
  }
  if (uploadInFlight) return;
  uploadInFlight = true;
  clearError();
  setFetchLoading(true);
  startAnalyzing();

  try {
    var formData = new FormData();
    for (var i = 0; i < files.length; i++) {
      formData.append('files', files[i]);
      formData.append('paths[]', files[i].webkitRelativePath);
    }

    var response = await fetchWithTimeout('/api/upload-folder', {
      method: 'POST',
      body: formData,
      // NAO definir Content-Type — browser define automaticamente com boundary
    }, 60000); // timeout mais generoso para uploads grandes

    var data = await response.json();

    if (!response.ok) {
      showError(data.error || 'Erro ao processar pasta. Verifique se contem index.html.');
      return;
    }

    state.fetchedHtml        = data.html;
    state.fetchedUrl         = '';
    state.uploadSessionId    = data.uploadSessionId;
    state.checkoutLinks      = (data.summary && data.summary.checkoutLinks) || [];
    state.delaySeconds       = (data.summary && data.summary.delaySeconds != null) ? data.summary.delaySeconds : null;
    state.hasDelay           = !!(data.summary && data.summary.hasDelay);
    state.delayScriptContent = (data.summary && data.summary.delayScriptContent) || null;
    state.delayType          = (data.summary && data.summary.delayType) || null;
    state.bundleImages       = (data.summary && data.summary.bundleImages) || {};

    var scriptsRemoved = (data.summary && data.summary.scriptsRemoved) || 0;
    var vslDetected    = (data.summary && data.summary.vslDetected)    || false;
    var linksFound     = state.checkoutLinks.length;

    valScripts.textContent = scriptsRemoved;
    valLinks.textContent   = linksFound;

    if (vslDetected) {
      valVsl.textContent = 'Sim';
      summaryVslItem.classList.remove('muted');
    } else {
      valVsl.textContent = 'Nao';
      summaryVslItem.classList.add('muted');
    }

    checkoutCountBadge.textContent = linksFound;
    renderCheckoutInputs(state.checkoutLinks);

    // Reveal sections (same logic as URL fetch)
    sectionSummary.classList.remove('section-hidden');
    sectionHeader.classList.remove('section-hidden');
    sectionPlayer.classList.remove('section-hidden');
    sectionCheckout.classList.remove('section-hidden');

    // Reveal delay section
    if (sectionDelay) {
      if (state.hasDelay && state.delaySeconds !== null) {
        sectionDelay.classList.remove('section-hidden');
        if (delayInput) delayInput.value = state.delaySeconds;
        if (delayStatus) {
          delayStatus.textContent = 'Delay detectado: ' + state.delaySeconds + 's';
          delayStatus.className = 'delay-status detected';
        }
      } else {
        sectionDelay.classList.remove('section-hidden');
        if (delayInput) delayInput.value = '';
        if (delayStatus) {
          delayStatus.textContent = 'Nenhum delay detectado na pagina';
          delayStatus.className = 'delay-status not-detected';
        }
      }
    }

    // Reveal bundle images section
    var sectionBundle = document.getElementById('section-bundle-images');
    if (sectionBundle) {
      if (Object.keys(state.bundleImages).length > 0) {
        sectionBundle.classList.remove('section-hidden');
        if (typeof renderBundleImages === 'function') renderBundleImages(state.bundleImages);
      } else {
        sectionBundle.classList.remove('section-hidden');
      }
    }

    // Reveal extra scripts section
    var sectionExtraScripts = document.getElementById('section-extra-scripts');
    if (sectionExtraScripts) {
      sectionExtraScripts.classList.remove('section-hidden');
    }

    // Reveal export button
    setExportEnabled(true);

  } catch (err) {
    showError('Falha de conexao ao processar pasta. Verifique se o servidor esta rodando.');
  } finally {
    uploadInFlight = false;
    setFetchLoading(false);
    stopAnalyzing();
  }
});
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser tab toggle | Interacao puramente local, sem risco |
| Browser -> FormData construction | webkitRelativePath lido do FileList nativo — confiavel em uso normal, mas paths podem ser forjados via HTTP direto |
| FormData -> /api/upload-folder | Upload multipart; validacao de extensao e path feita no servidor (Plan 08-01) |
| state.uploadSessionId -> /api/export-zip | UUID armazenado no state JS do cliente; transmitido no payload de export |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-08 | Spoofing | FormData Content-Type | mitigate | NAO definir Content-Type manualmente no fetch — browser define com boundary correto. Verificado no code review |
| T-08-09 | Information Disclosure | folder selection | accept | webkitdirectory mostra lista de arquivos no label, mas nao expoe conteudo ao servidor ate o POST. Risco negligivel |
| T-08-10 | Denial of Service | upload de pasta muito grande | mitigate | Timeout de 60s no fetchWithTimeout + limites do multer no servidor (10MB/file, 200 files) |
</threat_model>
