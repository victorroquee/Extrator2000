function t(key, replacements) {
  var str = (T[LANG] && T[LANG][key]) || (T['pt-BR'] && T['pt-BR'][key]) || key;
  if (replacements) {
    Object.keys(replacements).forEach(function(k) {
      str = str.replace('{' + k + '}', replacements[k]);
    });
  }
  return str;
}

function applyLang() {
  // Update html lang attribute
  document.documentElement.lang = (LANG === 'en') ? 'en' : 'pt-BR';

  // Apply text to all data-i18n elements
  document.querySelectorAll('[data-i18n]').forEach(function(el) {
    var key = el.getAttribute('data-i18n');
    el.innerHTML = t(key);
  });

  // Apply placeholders
  document.querySelectorAll('[data-i18n-placeholder]').forEach(function(el) {
    el.placeholder = t(el.getAttribute('data-i18n-placeholder'));
  });

  // Apply titles
  document.querySelectorAll('[data-i18n-title]').forEach(function(el) {
    el.title = t(el.getAttribute('data-i18n-title'));
  });

  // Apply aria-labels
  document.querySelectorAll('[data-i18n-aria]').forEach(function(el) {
    el.setAttribute('aria-label', t(el.getAttribute('data-i18n-aria')));
  });

  // Update flag toggle active state
  var toggle = document.getElementById('lang-toggle');
  if (toggle) {
    toggle.querySelectorAll('button').forEach(function(btn) {
      btn.classList.toggle('active', btn.getAttribute('data-lang') === LANG);
    });
  }

  // Update ANALYZING_STEPS from translations
  ANALYZING_STEPS = [
    t('analyzing.step1'), t('analyzing.step2'), t('analyzing.step3'),
    t('analyzing.step4'), t('analyzing.step5'), t('analyzing.step6')
  ];
}
